Data Base: 

CREATE TABLE users (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(100),
  email VARCHAR(100),
  password VARCHAR(255),
  role VARCHAR(20)
);

CREATE TABLE events (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  title VARCHAR(100),
  description TEXT,
  date DATE,
  location VARCHAR(100),
  created_by BIGINT
);

CREATE TABLE registrations (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  user_id BIGINT,
  event_id BIGINT,
  registration_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
